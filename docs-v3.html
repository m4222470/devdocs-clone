إعدادات التكوين

ملف الإعدادات

يمكنك تخصيص الموقع من خلال ملف الإعدادات.

{
  "site": {
    "title": "My Documentation",
    "description": "API Documentation",
    "url": "https://docs.example.com"
  },
  "theme": {
    "primaryColor": "#6366F1",
    "darkMode": true
  }
}
